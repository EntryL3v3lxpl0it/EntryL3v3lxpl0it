```bash
#!/usr/bin/env bash
#
# engagement-init.sh — Penetration Testing Methodology Suite scaffolder
#
# Creates the engagement folder tree for a chosen methodology, seeds scope /
# authorization / README / change-log stubs, drops in the canonical finding
# template + evidence conventions + a hashing helper.
#
# Master Index: 00_Methodology_Suite_Master_Index.md
# Source of truth for templates: ./templates/ (next to this script).
#
# Usage:
#   ./engagement-init.sh <methodology> <CLIENT> [--date YYYYMMDD] [--dir PATH]
#   ./engagement-init.sh --list
#   ./engagement-init.sh -h | --help
#
# Methodology keys:
#   ext   External Network            (EXT-)
#   int   Internal / Assumed-Breach   (INT-)   [state-creating]
#   ad    AD / Windows Enterprise     (AD-)    [state-creating]
#   web   Web Application             (WEB-)
#   api   API / Microservices         (API-)
#   saas  SaaS / Cloud-Exposed        (SAAS-)
#   mob   Mobile Application          (MOB-)
#   se    Social Engineering          (SE-)    [special auth + data-handling]
#   fuzz  Fuzzing                     (FUZZ-)
#
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
TEMPLATES_DIR="$SCRIPT_DIR/templates"

# ----------------------------------------------------------------------------
usage() {
  sed -n '2,30p' "$0" | sed 's/^# \{0,1\}//'
  exit "${1:-0}"
}

list_methodologies() {
  cat <<'EOF'
Methodology keys (prefix):
  ext    External Network             EXT-
  int    Internal / Assumed-Breach    INT-    [state-creating: change-log]
  ad     AD / Windows Enterprise      AD-     [state-creating: change-log]
  web    Web Application              WEB-
  api    API / Microservices          API-
  saas   SaaS / Cloud-Exposed         SAAS-
  mob    Mobile Application           MOB-
  se     Social Engineering           SE-     [exec+HR+Legal auth; data-handling]
  fuzz   Fuzzing                      FUZZ-
EOF
  exit 0
}

# ----------------------------------------------------------------------------
# Argument parsing
[ $# -ge 1 ] || usage 1
case "${1:-}" in
  -h|--help) usage 0 ;;
  --list)    list_methodologies ;;
esac
[ $# -ge 2 ] || { echo "error: need <methodology> and <CLIENT>"; usage 1; }

METH="$1"; CLIENT="$2"; shift 2
DATE="$(date -u +%Y%m%d)"
BASEDIR="."
while [ $# -gt 0 ]; do
  case "$1" in
    --date) DATE="$2"; shift 2 ;;
    --dir)  BASEDIR="$2"; shift 2 ;;
    *) echo "error: unknown option '$1'"; usage 1 ;;
  esac
done

# sanitize client for filesystem
CLIENT_SAFE="$(printf '%s' "$CLIENT" | tr ' /' '__' | tr -cd 'A-Za-z0-9_.-')"

# ----------------------------------------------------------------------------
# Per-methodology definition: PREFIX, STATE-CREATING, LOOT dir, and the exact
# authored directory tree (brace expansion pre-resolved for portability).
PREFIX=""; STATE_CREATING=0; LOOT=""; SPECIAL_AUTH=0; EVIDENCE_DIR="07-evidence"
declare -a DIRS=()

case "$METH" in
  ext)
    PREFIX="EXT"; LOOT="99-loot"
    DIRS=(
      00-scope 01-osint 02-discovery
      03-enum/web 03-enum/smb 03-enum/smtp 03-enum/snmp 03-enum/dns 03-enum/vpn 03-enum/db 03-enum/other
      04-vulnscan 05-validation 06-access
      07-evidence/screenshots 07-evidence/pcaps 07-evidence/raw
      08-report "$LOOT"
    ) ;;
  int)
    PREFIX="INT"; STATE_CREATING=1; LOOT="99-loot"
    DIRS=(
      00-scope 01-foothold 02-discovery 03-unauth
      04-ad-enum/bloodhound 04-ad-enum/ldap 04-ad-enum/shares
      05-credaccess/hashes 05-credaccess/tickets 05-credaccess/dpapi 05-credaccess/cracked
      06-privesc/windows 06-privesc/linux
      07-lateral 08-adpaths 09-objective
      10-cleanup/change-log 10-cleanup/reverted 11-report
      07-evidence/screenshots 07-evidence/raw "$LOOT"
    ) ;;
  ad)
    PREFIX="AD"; STATE_CREATING=1; LOOT="99-loot"
    DIRS=(
      00-scope 01-recon/bloodhound 01-recon/topology 01-recon/tiering
      02-credexposure 03-acl-paths 04-delegation 05-adcs 06-gpo
      07-entsvc/mssql 07-entsvc/sccm 07-entsvc/exchange 07-entsvc/wsus 07-entsvc/shares
      08-trusts 09-hybrid-entra 10-dominance 11-detection
      12-cleanup/change-log 12-cleanup/reverted 13-report
      07-evidence/screenshots 07-evidence/raw "$LOOT"
    ) ;;
  web)
    PREFIX="WEB"; LOOT="99-loot"
    DIRS=(
      00-scope 01-recon
      02-mapping/content 02-mapping/params 02-mapping/apis 02-mapping/js
      03-config 04-authn 05-authz
      06-injection/sqli 06-injection/xss 06-injection/cmdi 06-injection/ssti 06-injection/ssrf
      06-injection/xxe 06-injection/fileinc 06-injection/deser 06-injection/other
      07-logic 08-api
      09-source-review/sources-sinks 09-source-review/findings
      10-exploit 11-report
      07-evidence/burp 07-evidence/screenshots 07-evidence/raw 07-evidence/har "$LOOT"
    ) ;;
  api)
    PREFIX="API"; LOOT="99-loot"
    DIRS=(
      00-scope 01-arch/schemas 01-arch/inventory 01-arch/trust-map
      02-authn 03-authz 04-input
      05-microsvc/ssrf 05-microsvc/gateway 05-microsvc/intersvc 05-microsvc/admin-apis 05-microsvc/async 05-microsvc/secrets
      06-resource-logic
      07-infra/containers 07-infra/k8s 07-infra/cicd 07-infra/cloud
      08-source-review 09-exploit 10-report
      07-evidence/burp 07-evidence/screenshots 07-evidence/raw "$LOOT"
    ) ;;
  saas)
    PREFIX="SAAS"; LOOT="99-loot"
    DIRS=(
      00-scope 01-discovery/assets 01-discovery/storage 01-discovery/dns 01-discovery/fingerprint
      02-identity 03-tenant-isolation 04-authz 05-appsec
      06-cloud-substrate/imds 06-cloud-substrate/iam 06-cloud-substrate/storage 06-cloud-substrate/serverless 06-cloud-substrate/secrets
      07-saas-logic 08-integrations 09-takeover 10-exploit 11-report
      07-evidence/burp 07-evidence/screenshots 07-evidence/raw "$LOOT"
    ) ;;
  mob)
    PREFIX="MOB"; LOOT="99-loot"
    DIRS=(
      00-scope 01-recon
      02-static/decompiled 02-static/manifest 02-static/secrets 02-static/native 02-static/sdk
      03-storage 04-crypto 05-auth 06-network 07-platform-ipc 08-runtime 09-resilience 10-backend 11-privacy 12-report
      07-evidence/screenshots 07-evidence/traffic 07-evidence/raw "$LOOT"
    ) ;;
  se)
    PREFIX="SE"; STATE_CREATING=1; SPECIAL_AUTH=1; LOOT="99-secure-store"
    DIRS=(
      00-scope 01-osint 02-design 03-infra 04-prelaunch 05-execution 06-metrics
      07-vishing-physical 08-awareness 09-report 10-data-handling
      07-evidence/screenshots 07-evidence/raw "$LOOT"
    ) ;;
  fuzz)
    PREFIX="FUZZ"; LOOT="99-artifacts"
    DIRS=(
      00-scope 01-target-analysis 02-harness
      03-build/afl 03-build/cmplog 03-build/libfuzzer 03-build/asan 03-build/ubsan 03-build/msan 03-build/coverage
      04-corpus/seeds 04-corpus/minimized 04-corpus/dict
      05-campaign/out 05-campaign/logs 05-campaign/coverage
      06-crashes/raw 06-crashes/unique 06-crashes/minimized
      07-triage 08-patch-validation 09-report
      07-evidence/screenshots 07-evidence/raw "$LOOT"
    ) ;;
  *)
    echo "error: unknown methodology '$METH'"; echo; list_methodologies ;;
esac

# ----------------------------------------------------------------------------
ROOT="$BASEDIR/${PREFIX}-${DATE}-${CLIENT_SAFE}"
if [ -e "$ROOT" ]; then
  echo "error: '$ROOT' already exists — refusing to overwrite."; exit 1
fi

echo "[*] Scaffolding $METH engagement: $ROOT"
for d in "${DIRS[@]}"; do mkdir -p "$ROOT/$d"; done
# every engagement gets a findings/ dir for per-finding Lifecycle records
mkdir -p "$ROOT/findings"

# ----------------------------------------------------------------------------
# Seed canonical templates (copy from templates/, else write a pointer note)
copy_template() {  # $1 = filename  $2 = dest path
  if [ -f "$TEMPLATES_DIR/$1" ]; then
    cp "$TEMPLATES_DIR/$1" "$2"
  else
    echo "[!] templates/$1 not found next to script — see the Master Index for the canonical version." > "$2"
  fi
}
copy_template FINDING-TEMPLATE.md     "$ROOT/findings/FINDING-TEMPLATE.md"
copy_template EVIDENCE-CONVENTIONS.md "$ROOT/$EVIDENCE_DIR/EVIDENCE-CONVENTIONS.md"

# ----------------------------------------------------------------------------
# Seed 00-scope/AUTHORIZATION.md
if [ "$SPECIAL_AUTH" -eq 1 ]; then
  cat > "$ROOT/00-scope/AUTHORIZATION.md" <<'EOF'
# Authorization — Social Engineering (EXPANDED, hard blocker)

> Targets are PEOPLE. Do not proceed past this file until ALL are satisfied.

## Authorization chain (all required)
- [ ] Executive sponsor (authority over the targeted population) — signed
- [ ] HR review / approval — documented
- [ ] Legal review / approval — documented
- [ ] (External testers) contract + insurance + get-out-of-jail letter

## Legal clearance (jurisdiction-specific — confirm with counsel)
- [ ] Pretexting law / GLBA pretexting (financial info)
- [ ] Recording consent (one-/two-party) for vishing
- [ ] Privacy law (GDPR/CCPA) for employee PII
- [ ] PROHIBITED: impersonating government / law enforcement / regulators
- [ ] PROHIBITED: impersonating a real named third party without consent

## Ethical guardrails / duty of care
- [ ] Prohibited-pretext list honored (no layoffs/bonuses/death/health/disaster/protected-class)
- [ ] Vulnerable-individual exclusions applied
- [ ] Blame-free immediate teaching moment for anyone who interacts
- [ ] Metrics aggregated/anonymized — NO named individuals, no discipline/ranking

## Deconfliction
- White cell (in-the-know): ______________________
- Kill switch / stop procedure: __________________
- IR deconfliction decision (read-in? yes/no): ____
- Emergency contacts: ____________________________
EOF
else
  cat > "$ROOT/00-scope/AUTHORIZATION.md" <<'EOF'
# Authorization (hard blocker — Phase 0 does not pass without this)

- [ ] Signed authorization / get-out-of-jail letter (testing entity, authorizing party, ownership attestation)
- [ ] In-scope targets (domains / IPs / CIDRs / hosts / apps / accounts)
- [ ] EXCLUSIONS (assets/paths/accounts/fragile systems out of scope)
- [ ] Rules of Engagement (intensity, prohibited techniques, windows, data handling)
- [ ] Source IPs for allow-listing / blue-team deconfliction
- [ ] Lockout policy + spray budget (if auth testing)
- [ ] Cloud/provider testing policy honored (if applicable)
- [ ] Emergency stop procedure + contacts

Authorizing party: ____________________   Date: __________
Testing entity:    ____________________
Engagement window: ____________________
EOF
fi

# Seed 00-scope/SCOPE.md
cat > "$ROOT/00-scope/SCOPE.md" <<EOF
# Scope — ${PREFIX} engagement (${CLIENT})

Methodology: ${METH}    Prefix: ${PREFIX}    Created (UTC): ${DATE}

## In scope
- 

## Out of scope / exclusions
- 

## Rules of Engagement
- Environment (prod/staging):
- Intensity / rate limits:
- Destructive testing allowed? :
- Persistence / dominance allowed? (gated):
- Data handling (real vs marker):

## Identities / accounts provided
- 
EOF

# ----------------------------------------------------------------------------
# Seed change-log for state-creating engagements
if [ "$STATE_CREATING" -eq 1 ]; then
  CL_DIR="$ROOT/10-cleanup/change-log"
  [ "$METH" = "ad" ] && CL_DIR="$ROOT/12-cleanup/change-log"
  [ "$METH" = "se" ] && CL_DIR="$ROOT/10-data-handling"
  mkdir -p "$CL_DIR"
  cat > "$CL_DIR/changes.md" <<'EOF'
# Change Log (mandatory — maintain from the FIRST modification)

Every object created/changed during testing is logged here and reverted before close.
For SE engagements this also tracks captured data + its destruction.

| # | UTC time | What (object/action) | Where (host/dir/system) | Why | Reverted? | Revert evidence |
|---|----------|----------------------|-------------------------|-----|-----------|-----------------|
| 1 |          |                      |                         |     | [ ]       |                 |

## Examples of things that MUST be logged & reverted
- Added computer accounts (RBCD), group memberships, ACL/owner edits, AdminSDHolder
- Created/linked GPOs + SYSVOL artifacts; issued certificates
- Dropped agents/binaries/tasks/services; forged tickets; captured caches
- (SE) Harvested credentials / PII — protect then DESTROY (log destruction)
- Campaign domains / infra to decommission
EOF
fi

# ----------------------------------------------------------------------------
# Seed hash-evidence.sh helper
cat > "$ROOT/hash-evidence.sh" <<EOF
#!/usr/bin/env bash
# Hash all raw evidence into $EVIDENCE_DIR/SHA256SUMS (run from engagement root).
set -euo pipefail
cd "\$(dirname "\$0")"
EVID="$EVIDENCE_DIR"
find "\$EVID" -type f ! -name SHA256SUMS -exec sha256sum {} \\; | sort -k2 > "\$EVID/SHA256SUMS"
echo "[*] Wrote \$EVID/SHA256SUMS (\$(wc -l < "\$EVID/SHA256SUMS") files)"
echo "    Verify later with: sha256sum -c \$EVID/SHA256SUMS"
EOF
chmod +x "$ROOT/hash-evidence.sh" 2>/dev/null || true

# ----------------------------------------------------------------------------
# Seed engagement README
cat > "$ROOT/README.md" <<EOF
# ${PREFIX}-${DATE}-${CLIENT_SAFE}

Methodology: **${METH}** — see the matching playbook and the Master Index
(\`00_Methodology_Suite_Master_Index.md\`).

## Conventions (see $EVIDENCE_DIR/EVIDENCE-CONVENTIONS.md)
- Evidence naming: \`<phase>_<target>_<tool>_<UTC>.<ext>\`
- Hash raw evidence: run \`./hash-evidence.sh\` (writes 07-evidence/SHA256SUMS)
- Keep evidence (observed) separate from hypothesis (inferred)
- One Lifecycle record per finding: \`cp findings/FINDING-TEMPLATE.md findings/F-001-<slug>.md\`
- Close every finding through the Vulnerability Lifecycle playbook (V)

## Workflow
1. Complete 00-scope/AUTHORIZATION.md and 00-scope/SCOPE.md (Phase 0 hard blocker).
2. Work the numbered phases in order; honor validation gates.
3. Route every confirmed candidate to findings/ via the Lifecycle.
$( [ "$STATE_CREATING" -eq 1 ] && echo "4. Log EVERY created object in the change-log; revert all before close." )

## Tree
$(for d in "${DIRS[@]}"; do echo "- $d/"; done)
- findings/
EOF

# ----------------------------------------------------------------------------
echo "[*] Done."
echo "    Root:        $ROOT"
echo "    Next:        edit 00-scope/AUTHORIZATION.md and 00-scope/SCOPE.md"
echo "    Findings:    cp $ROOT/findings/FINDING-TEMPLATE.md $ROOT/findings/F-001-<slug>.md"
[ "$STATE_CREATING" -eq 1 ] && echo "    Change-log:  maintain it from your first modification."
echo "    Hashing:     run $ROOT/hash-evidence.sh before reporting."
```
